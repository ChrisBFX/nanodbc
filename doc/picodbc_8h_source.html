523  template<class T>
524  T get(short column, unsigned long row = 0)
525  {
526  detail::bound_column& col = result_impl_get_bound_column(impl_, column, row);
527  switch(col.ctype_)
528  {
529  case SQL_C_SSHORT: return (T)*(short*)(col.pdata_ + row * col.clen_);
530  case SQL_C_USHORT: return (T)*(unsigned short*)(col.pdata_ + row * col.clen_);
531  case SQL_C_SLONG: return (T)*(long*)(col.pdata_ + row * col.clen_);
532  case SQL_C_ULONG: return (T)*(unsigned long*)(col.pdata_ + row * col.clen_);
533  case SQL_C_FLOAT: return (T)*(float*)(col.pdata_ + row * col.clen_);
534  case SQL_C_DOUBLE: return (T)*(double*)(col.pdata_ + row * col.clen_);
535  case SQL_C_CHAR: if (!col.blob_) return detail::convert<T>((col.pdata_ + row * col.clen_));
536  }
537  throw type_incompatible_error();
538  }

565 template<>
566 inline std::string result::get<std::string>(short column, unsigned long row)
567 {
568  detail::bound_column& col = result_impl_get_bound_column(impl_, column, row);
569  char buffer[1024];
570  switch(col.ctype_)
571  {
572  case SQL_C_CHAR:
573  case SQL_C_GUID:
574  case SQL_C_BINARY:
575  if(col.blob_)
576  throw std::runtime_error("blob not implemented yet"); // #T implement load_blob(column);
577  return (col.pdata_ + row * col.clen_);
578 
579  case SQL_C_LONG:
580  if(std::snprintf(buffer, sizeof(buffer), "%ld", *(long*)(col.pdata_ + row * col.clen_)) != 1)
581  throw type_incompatible_error();
582  return buffer;
583 
584  case SQL_C_DOUBLE:
585  if(std::snprintf(buffer, sizeof(buffer), "%lf", *(double*)(col.pdata_ + row * col.clen_)) != 1)
586  throw type_incompatible_error();
587  return buffer;
588 
589  case SQL_C_DATE:
590  throw std::runtime_error("date not implemented yet"); // #T implement data conversion
591 
592  case SQL_C_TIMESTAMP:
593  throw std::runtime_error("timestamp not implemented yet"); // #T implement timestamp conversion
594  }
595  throw type_incompatible_error();
596 }
