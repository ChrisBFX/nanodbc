509  template<class T>
510  T get(short column)
511  {
512  detail::bound_column& col = result_impl_get_bound_column(impl_, column);
513  switch(col.ctype)
514  {
515  case SQL_C_SSHORT: return (T)*(short*)col.pdata;
516  case SQL_C_USHORT: return (T)*(unsigned short*)col.pdata;
517  case SQL_C_SLONG: return (T)*(long*)col.pdata;
518  case SQL_C_ULONG: return (T)*(unsigned long *)col.pdata;
519  case SQL_C_FLOAT: return (T)*(float*)col.pdata;
520  case SQL_C_DOUBLE: return (T)*(double*)col.pdata;
521  case SQL_C_CHAR: if (!col.blob) return detail::convert<T>(col.pdata);
522  }

556 inline std::string result::get<std::string>(short column)
557 {
558  detail::bound_column& col = result_impl_get_bound_column(impl_, column);
559  char buffer[1024];
560  switch(col.ctype)
561  {
562  case SQL_C_CHAR:
563  case SQL_C_GUID:
564  case SQL_C_BINARY:
565  if(col.blob)
566  throw std::runtime_error("blob not implemented yet"); // #T implement load_blob(column);
567  return col.pdata;
568 
569  case SQL_C_LONG:
570  if(std::snprintf(buffer, sizeof(buffer), "%ld", *(long*)col.pdata) != 1)
571  throw type_incompatible_error();
572  return buffer;
573 
574  case SQL_C_DOUBLE:
575  if(std::snprintf(buffer, sizeof(buffer), "%lf", *(double*)col.pdata) != 1)
576  throw type_incompatible_error();
577  return buffer;

579  case SQL_C_DATE:
