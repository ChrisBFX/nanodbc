772 inline std::string statement::get_direct<std::string>(long column, bool error_on_null)
773 {
774  SQLLEN cb_needed = 0;
775  char small_buff[256];
776  RETCODE rc = SQLGetData(stmt_, column + 1, SQL_C_CHAR, small_buff, sizeof(small_buff), &cb_needed);
777  if(detail::is_null(cb_needed))
778  null_[column] = 1;
779  if(detail::is_null(cb_needed) && error_on_null)
780  {
781  throw null_access_error();
782  }
783  else if(!detail::success(rc))
784  {
785  picoDBC_THROW_DATABASE_ERROR(stmt_, SQL_HANDLE_STMT);
786  }
787  else if(detail::success(rc) && cb_needed < sizeof(small_buff))
788  {
789  return std::string(small_buff);
790  }
791  else if(cb_needed > 0)
792  {
793  SQLLEN sz_buff = cb_needed + 1;
794  char p_data[sz_buff];
795  std::size_t offset = sizeof(small_buff) - 1 ;
796  std::memcpy(p_data, small_buff, offset);
797  rc = SQLGetData(stmt_, column + 1, SQL_C_CHAR, (SQLCHAR*)(p_data + offset), sz_buff - offset, &cb_needed);
798  if (!detail::success(rc))
799  picoDBC_THROW_DATABASE_ERROR(stmt_, SQL_HANDLE_STMT);
800  return std::string(p_data);
801  }
802  return std::string();
